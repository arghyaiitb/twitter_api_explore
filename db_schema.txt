
Tweets:
    id : BigIntegerField(primary_key=True)  # tweet id in string is unique
    tweet_search :CharField(max_length=280)  # the string on which the search function was executed (generated by me)
    full_text : TextField() # the full tweet text
    created_at : DateTimeField()  # Date time when the tweet was created
    author_id : BigIntegerField()  # match user table id data to tweet
    author_screen_name : CharField(max_length=100,default=None,null=True) #screen name used
    favorite_count : IntegerField(default=0)  # likes on the tweet
    lang : CharField(max_length=10)  # language of the tweet
    geo_location : JSONField(default=None, null=True)  # location tagged in the tweet
    retweet_count : IntegerField()  # total retweet on the tweet
    hashtags : ArrayField(CharField(max_length=280), default=[], null=True, blank=True)  # all the hastags used
    user_mentions : ArrayField(CharField(max_length=280), default=[], null=True, blank=True)  # all the hastags used  # all the users tages using @xyz
    urls : ArrayField(TextField(), default=[], null=True, blank=True)  # all the hastags used  # all the urls linked to the tweet
    tweet_score : FloatField(default=0.0) # 1.0


Users
    id : BigIntegerField(primary_key=True)  # user id of a person
    location : CharField(max_length=100, default=None, null=True, blank=True)  # the location of the user
    name : CharField(max_length=200, default=None, null=True, blank=True)  # name of the user
    description : TextField(default=None, null=True, blank=True)  # description of the user / like bio
    screen_name : CharField(max_length=100, default=None, null=True, blank=True)  # unique user name for using @user name
    verified : BooleanField(default=False)  # to see if the user is verified or not
    created_at : DateTimeField()  # when the user was created
    favourites_count : IntegerField(default=0)  # number of likes
    followers_count : IntegerField(default=0)  # people following this user
    friends_count : IntegerField(default=0)  # the number of people this user is following
    lang : CharField(max_length=20, default=None, null=True) # english
    statuses_count : IntegerField(default=0)  # total no. of tweets done by the user
